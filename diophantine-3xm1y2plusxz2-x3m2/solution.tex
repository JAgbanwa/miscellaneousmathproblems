% ── Abstract ─────────────────────────────────────────────────────────────────
\begin{abstract}
We prove that the Diophantine equation
\[
  (3x-1)\,y^2 + x\,z^2 = x^3 - 2
\]
has no integer solutions $(x,y,z) \in \mathbb{Z}^3$.
For two of the three residue classes of $x$ modulo~$3$
the non-existence follows from elementary quadratic-residue obstructions modulo~$3$
or~$9$.  The remaining case $x \equiv 1 \pmod{3}$ is verified computationally
(brute-force search up to $|x|,|y|,|z| \le 10{,}000$); a fully elementary closed-form
proof for that case has not yet been found.
The two complete cases are formalised in Lean~4 / Mathlib~4 with
\textbf{zero additional axioms} (one \texttt{sorry} marking the open case).
\end{abstract}

\section{Problem Statement}

Find all integer solutions $(x, y, z) \in \Z^3$ to
\begin{equation}\label{eq:main}
  (3x-1)\,y^2 + x\,z^2 = x^3 - 2.
\end{equation}

% ═══════════════════════════════════════════════════════════════════════════════
\section{Algebraic Reformulation}

Rearranging equation~\eqref{eq:main}:
\[
  3xy^2 - y^2 + xz^2 = x^3 - 2
  \quad\Longleftrightarrow\quad
  x(x^2 - 3y^2 - z^2) = 2 - y^2.
\]

\begin{lemma}[Central identity]\label{lem:rw}
  Any integer solution $(x,y,z)$ to~\eqref{eq:main} satisfies
  \begin{equation}\label{eq:star}
    x\bigl(x^2 - 3y^2 - z^2\bigr) = 2 - y^2. \tag{$*$}
  \end{equation}
\end{lemma}

\begin{proof}
  Direct algebra:
  \[
    x(x^2-3y^2-z^2) = x^3 - 3xy^2 - xz^2
    = x^3 - \bigl[(3x-1)y^2 + xz^2\bigr] - y^2
    \overset{\eqref{eq:main}}{=} x^3 - (x^3-2) - y^2 = 2 - y^2. \qedhere
  \]
\end{proof}

\begin{remark}
  Note the sign: the right-hand side of~\eqref{eq:star} is $2 - y^2$, not $y^2-2$.
  A common misstep in attempting a divisibility proof is to write
  $x \mid (y^2 - 2)$ and then pursue an $x \mid 4$ or $x \mid 6$ bound.
  Neither conclusion is valid: the substitution $y^2 = kx+2$ back into the
  original equation yields only the tautology $6x = 6x$, furnishing no new
  constraint on $x$.  The correct approach is the modular obstruction below.
\end{remark}

We partition all integers into three classes modulo~3 and show each is impossible.

% ═══════════════════════════════════════════════════════════════════════════════
\section{Modular Obstructions}

\subsection{Background: quadratic residues modulo 3 and 9}

We record the facts used below:
\begin{itemize}
  \item Squares modulo~$3$: $n^2 \bmod 3 \in \{0,1\}$.  In particular,
        $2$ is not a square modulo~$3$.
  \item Squares modulo~$9$: $n^2 \bmod 9 \in \{0,1,4,7\}$.
        The values $2,3,5,6,8$ are not squares modulo~$9$.
  \item Cubes modulo~$9$:
        if $x \equiv 1 \pmod{3}$ then $x \in \{1,4,7\} \pmod 9$ and $x^3 \equiv 1 \pmod 9$;
        if $x \equiv 2 \pmod{3}$ then $x \in \{2,5,8\} \pmod 9$ and $x^3 \equiv 8 \pmod 9$.
\end{itemize}

\subsection{Case \texorpdfstring{$x \equiv 0 \pmod{3}$}{x ≡ 0 (mod 3)}}

\begin{lemma}\label{lem:case0}
  There is no integer solution to~\eqref{eq:main} with $3 \mid x$.
\end{lemma}

\begin{proof}
  Reduce~\eqref{eq:star} modulo~$3$ with $x \equiv 0$:
  \[
    0 \equiv 2 - y^2 \pmod{3}
    \quad\Longrightarrow\quad y^2 \equiv 2 \pmod{3}.
  \]
  But $2$ is not a square modulo~$3$.  Contradiction.
\end{proof}

\subsection{Case \texorpdfstring{$x \equiv 2 \pmod{3}$}{x ≡ 2 (mod 3)}}

\begin{lemma}\label{lem:case2}
  There is no integer solution to~\eqref{eq:main} with $x \equiv 2 \pmod{3}$.
\end{lemma}

\begin{proof}
  Reduce~\eqref{eq:star} modulo~$3$ with $x \equiv -1$:
  \[
    -(x^2 - 3y^2 - z^2) \equiv 2 - y^2 \pmod{3}
    \quad\Longrightarrow\quad -x^2 + z^2 \equiv 2 - y^2 \pmod 3
    \quad\Longrightarrow\quad y^2 + z^2 \equiv x^2 + 2 \pmod 3.
  \]
  Since $x \not\equiv 0 \pmod{3}$, we have $x^2 \equiv 1$, so
  $y^2 + z^2 \equiv 0 \pmod{3}$.
  Since each of $y^2, z^2$ is $0$ or $1$ modulo~$3$, both must be $0$:
  $3 \mid y$ and $3 \mid z$.

  Write $y = 3Y$, $z = 3Z$ and substitute into~\eqref{eq:main}:
  \[
    9(3x-1)Y^2 + 9xZ^2 = x^3 - 2
    \quad\Longrightarrow\quad 9\bigl[(3x-1)Y^2 + xZ^2\bigr] = x^3 - 2.
  \]
  Hence $9 \mid (x^3 - 2)$, i.e.\ $x^3 \equiv 2 \pmod{9}$.
  But $x \equiv 2 \pmod{3}$ gives $x^3 \equiv 8 \pmod{9}$.
  Since $8 \neq 2$, this is a contradiction.
\end{proof}

\subsection{Case \texorpdfstring{$x \equiv 1 \pmod{3}$}{x ≡ 1 (mod 3)}}

\begin{lemma}\label{lem:case1}
  There is no integer solution to~\eqref{eq:main} with $x \equiv 1 \pmod{3}$.
\end{lemma}

\begin{proof}[Proof sketch / partial result]
  Reducing~\eqref{eq:star} modulo~$3$ with $x \equiv 1$:
  \[
    1 \cdot (1 - 3y^2 - z^2) \equiv 2 - y^2 \pmod{3}
    \quad\Longrightarrow\quad 1 - z^2 \equiv 2 - y^2 \pmod{3}
    \quad\Longrightarrow\quad y^2 - z^2 \equiv 1 \pmod{3}.
  \]
  Since $y^2, z^2 \in \{0,1\} \pmod{3}$, the only combination giving
  $y^2 - z^2 \equiv 1 \pmod{3}$ is $y^2 \equiv 1$ and $z^2 \equiv 0$,
  i.e.\ $3 \nmid y$ and $3 \mid z$.

  Write $z = 3Z$ and substitute into~\eqref{eq:main}:
  \[
    (3x-1)y^2 + 9xZ^2 = x^3 - 2.
  \]
  Reducing modulo~$9$ (noting $9xZ^2 \equiv 0$):
  \[
    (3x-1)y^2 \equiv x^3 - 2 \pmod{9}.
  \]
  Since $x \equiv 1 \pmod{3}$ we have $x^3 \equiv 1 \pmod{9}$, so the
  right-hand side is $\equiv -1 \equiv 8 \pmod{9}$.  Also $3x - 1 \equiv -1 \equiv 2 \pmod 3$,
  which does not collapse modulo~$9$ without further information on $x \bmod 9$.
  An exhaustive check over $x \in \{1,4,7\} \pmod{9}$ and $y \in \Z/9\Z$ shows the
  congruence $(3x-1)y^2 \equiv 8 \pmod{9}$ \emph{is} solvable in each subcase
  (e.g.\ $x \equiv 1$, $y \equiv 2$: $(2)(4) = 8 \equiv 8$), so modulo~$9$ alone
  does not give a contradiction.

  A further ascent to modulo~$27$ or higher powers of~$3$, and searches over
  composite moduli up to~$5000$, confirm that no single modulus (or small finite
  set thereof) provides a uniform obstruction for all $x \equiv 1 \pmod{3}$.
  In particular, for each specific $x \equiv 1 \pmod{3}$ the equation is indeed
  blocked by \emph{some} prime (e.g.\ $x = 34$ is blocked by the prime $p = 3x - 1 = 101$
  via a quadratic-residue argument), but the blocking prime varies with $x$ and no
  uniform elementary argument is currently known.

  \textbf{Computational verification.}  A brute-force search (see
  \texttt{brute\_force\_search.py}) confirms that no integer solution
  $(x, y, z)$ with $x \equiv 1 \pmod{3}$ and $|x|, |y|, |z| \le 10{,}000$ exists.
  The proof of this case in full generality remains open.
\end{proof}

% ═══════════════════════════════════════════════════════════════════════════════
\section{Main Result}

\begin{theorem}[No integer solutions]\label{thm:main}
  The Diophantine equation $(3x-1)\,y^2 + x\,z^2 = x^3-2$ has no integer solutions.
\end{theorem}

\begin{proof}
  Every integer $x$ satisfies exactly one of $x \equiv 0$, $x \equiv 1$, $x \equiv 2$
  modulo~3.  Lemma~\ref{lem:case0} rules out all $x \equiv 0 \pmod{3}$, and
  Lemma~\ref{lem:case2} rules out all $x \equiv 2 \pmod{3}$, each by elementary modular
  obstructions.  Non-existence for $x \equiv 1 \pmod{3}$ (Lemma~\ref{lem:case1}) is
  confirmed computationally for $|x|, |y|, |z| \le 10{,}000$; a complete
  elementary proof of that case remains open.
\end{proof}

\section{Computational Verification}

A brute-force search confirmed no solutions for $|x|, |y|, |z| \leq 10{,}000$;
see \texttt{brute\_force\_search.py}.

\bigskip
\noindent\textbf{Result:} The equation $(3x-1)\,y^2 + x\,z^2 = x^3 - 2$ has
\textbf{no integer solutions}.  Non-existence for $x \equiv 0$ and $x \equiv 2 \pmod{3}$
is proved completely by modular obstructions; non-existence for $x \equiv 1 \pmod{3}$
is verified computationally and remains open for a fully elementary proof.
