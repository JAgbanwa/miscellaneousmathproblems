\title{Completeness of the Full Three-Equation System}
\author{}
\date{}
\maketitle

\paragraph{System.}
For $a,r_2,r_3,g\in\mathbb{Q}$ with $r_2r_3\neq 0$, define
\[
b:=\frac{a^2-r_2^2}{2r_2},\quad
e:=\frac{a^2+r_2^2}{2r_2},\quad
c:=\frac{a^2-r_3^2}{2r_3},\quad
f:=\frac{a^2+r_3^2}{2r_3}.
\]
Then the system is
\[
a^2+b^2=e^2,\qquad
a^2+c^2=f^2,\qquad
b^2+c^2=g^2,
\]
equivalently
\[
a^2+\left(\frac{a^2-r_2^2}{2r_2}\right)^2
=
\left(\frac{a^2+r_2^2}{2r_2}\right)^2,
\]
\[
a^2+\left(\frac{a^2-r_3^2}{2r_3}\right)^2
=
\left(\frac{a^2+r_3^2}{2r_3}\right)^2,
\]
\[
\left(\frac{a^2-r_2^2}{2r_2}\right)^2
+
\left(\frac{a^2-r_3^2}{2r_3}\right)^2
=
g^2.
\]

\begin{proposition}[Soundness]
If $a,r_2,r_3,g\in\mathbb{Q}$ with $r_2r_3\neq 0$ satisfy
\[
\left(\frac{a^2-r_2^2}{2r_2}\right)^2
+
\left(\frac{a^2-r_3^2}{2r_3}\right)^2
=
g^2,
\]
then the derived variables $(a,b,e,c,f,g)$ satisfy
\[
a^2+b^2=e^2,\qquad
a^2+c^2=f^2,\qquad
b^2+c^2=g^2.
\]
\end{proposition}

\begin{proof}
The first two identities are polynomial identities in $(a,r_2)$ and $(a,r_3)$.
The third identity is exactly the displayed hypothesis after substituting $b,c$.
\end{proof}

\begin{theorem}[Completeness over $\mathbb{Q}$]
Let $a,b,e,c,f,g\in\mathbb{Q}$ satisfy
\[
a^2+b^2=e^2,\qquad
a^2+c^2=f^2,\qquad
b^2+c^2=g^2,
\]
and assume
\[
b<e,\qquad c<f.
\]
Then there exist nonzero $r_2,r_3\in\mathbb{Q}$ such that
\[
b=\frac{a^2-r_2^2}{2r_2},\qquad
e=\frac{a^2+r_2^2}{2r_2},\qquad
c=\frac{a^2-r_3^2}{2r_3},\qquad
f=\frac{a^2+r_3^2}{2r_3},
\]
and
\[
\left(\frac{a^2-r_2^2}{2r_2}\right)^2
+
\left(\frac{a^2-r_3^2}{2r_3}\right)^2
=
g^2.
\]
One may take
\[
r_2=e-b,\qquad r_3=f-c.
\]
\end{theorem}

\begin{proof}
From the first two equations and $b<e$, $c<f$, define
\[
r_2:=e-b,\qquad r_3:=f-c.
\]
Then $r_2,r_3\neq 0$, and the standard reconstruction gives
\[
b=\frac{a^2-r_2^2}{2r_2},\quad
e=\frac{a^2+r_2^2}{2r_2},\quad
c=\frac{a^2-r_3^2}{2r_3},\quad
f=\frac{a^2+r_3^2}{2r_3}.
\]
Substitute the first and third formulas into $b^2+c^2=g^2$ to get the final identity.
\end{proof}

\begin{theorem}[Integer-input corollary]
If $a,b,e,c,f,g\in\mathbb{Z}$ satisfy
\[
a^2+b^2=e^2,\qquad
a^2+c^2=f^2,\qquad
b^2+c^2=g^2,
\]
with $b<e$ and $c<f$, then there exist nonzero $r_2,r_3\in\mathbb{Q}$ satisfying
the same formulas above.
\end{theorem}

\begin{proof}
Embed the integer data into $\mathbb{Q}$ and apply the rational theorem.
\end{proof}

\paragraph{Conclusion.}
Using the same orientation assumptions as the Lean development, the full
three-equation system is complete in the variables $b,c$:
every rational or integer solution is captured by the divisor parameters
$r_2,r_3$.
