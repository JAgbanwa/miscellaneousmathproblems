\title{Completeness of the Divisor Parameterization for Integer Solutions}
\author{}
\date{}
\maketitle

Consider integer unknowns
\[
a,b,e,c,f\in\mathbb Z
\]
satisfying
\[
a^2+b^2=e^2,\qquad a^2+c^2=f^2,
\]
with assumptions
\[
b<e,\qquad c<f.
\]

\begin{theorem}
The system above is equivalent to the existence of nonzero rational parameters
\[
r_2,r_3\in\mathbb Q^\times
\]
such that
\[
b=\frac{a^2-r_2^2}{2r_2},\qquad
e=\frac{a^2+r_2^2}{2r_2},
\]
\[
c=\frac{a^2-r_3^2}{2r_3},\qquad
f=\frac{a^2+r_3^2}{2r_3}.
\]
In particular, every integer solution arises from this parameterization.
\end{theorem}

\begin{proof}
\textbf{(Soundness)}
For nonzero $r_2,r_3\in\mathbb Q$, define
\[
b:=\frac{a^2-r_2^2}{2r_2},\quad
e:=\frac{a^2+r_2^2}{2r_2},\quad
c:=\frac{a^2-r_3^2}{2r_3},\quad
f:=\frac{a^2+r_3^2}{2r_3}.
\]
Direct expansion gives
\[
a^2+b^2=e^2,\qquad a^2+c^2=f^2.
\]

\medskip
\textbf{(Completeness)}
Given integers with
\[
a^2+b^2=e^2,\qquad a^2+c^2=f^2,\qquad b<e,\qquad c<f,
\]
set
\[
r_2:=e-b,\qquad r_3:=f-c.
\]
Then $r_2,r_3\neq 0$ and
\[
a^2=(e-b)(e+b)=r_2(e+b),\qquad
a^2=(f-c)(f+c)=r_3(f+c).
\]
Solving each pair
\[
e-b=r_2,\ e+b=\frac{a^2}{r_2}
\quad\text{and}\quad
f-c=r_3,\ f+c=\frac{a^2}{r_3}
\]
yields
\[
b=\frac{a^2-r_2^2}{2r_2},\quad
e=\frac{a^2+r_2^2}{2r_2},\quad
c=\frac{a^2-r_3^2}{2r_3},\quad
f=\frac{a^2+r_3^2}{2r_3}.
\]
\end{proof}
