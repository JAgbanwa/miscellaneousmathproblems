\section*{Theorem}
There are no integers $x,y,z$ with $z$ even, $y$ odd, and $x\equiv 3\pmod 4$ satisfying
\[6 + x^{3} - y^{2} + y^{2}z^{2}=0.\]

\section*{Proof}
Rewrite the equation as
\begin{equation}\label{eq:main}
x^{3}+6 = -y^{2}(z^{2}-1).
\end{equation}

We split into four exhaustive cases.

\paragraph{Case 1: $y=0$.}
If $y = 0$ then \eqref{eq:main} reduces to $x^{3} = -6$. No integer cube
equals $-6$, because $(-2)^3 = -8$ and $(-1)^3 = -1$. Hence there is no
solution with $y=0$.

\paragraph{Case 2: $z=0$, $y\neq 0$.}
If $z = 0$ then \eqref{eq:main} becomes
\begin{equation}\label{eq:mordell}
y^{2} = x^{3} + 6.
\end{equation}
This is Mordell's equation with parameter $6$. It is a classical result that
this elliptic curve has no integer points. In particular, the curve with
LMFDB label \texttt{24.a3} has rank $0$ over $\mathbb Q$ and no rational
points besides the point at infinity. Therefore there is no integer solution
with $z=0$ and $y\neq 0$.

\paragraph{Case 3: $z=\pm 1$, $y\neq 0$.}
If $z = \pm 1$ then $z^{2}-1 = 0$ and \eqref{eq:main} reduces again to
$x^{3} = -6$. As in Case 1, this is impossible.

\paragraph{Case 4: $|z|\ge 2$, $y\neq 0$.}
Since $z$ is even, $z^{2}-1$ is an odd integer and $z^{2}-1 \ge 3$. Because
$y\neq 0$, the right-hand side of \eqref{eq:main} is at most $-3$. Thus
$6+x^{3} \le -3$, which forces $x^{3} \le -9$ and therefore $x \le -3$. Because
$x\equiv 3 \pmod 4$, in fact $x \le -5$. Set $X := -x\ge 5$, so
$X \equiv 1 \pmod 4$.

Rewrite \eqref{eq:main} as
\begin{equation}\label{eq:Xeq}
X^{3}+6 = y^{2}(z^{2}-1).
\end{equation}
The right-hand side is odd, so $X$ is odd.

Since $z$ is even, both $z-1$ and $z+1$ are odd and coprime. One of them is
congruent to $1$ modulo $4$ and the other is congruent to $3$ modulo $4$.
Because $y$ is odd, $y^{2}\equiv 1 \pmod 4$. Hence
\begin{equation}\label{eq:rhs-mod4}
\{(z-1)y^{2},\;(z+1)y^{2}\} \equiv \{1,3\} \pmod 4.
\end{equation}

On the other hand,
\begin{equation}\label{eq:mod4}
X^{3} + 6 \equiv 1 \pmod 4
\end{equation}
for $X \equiv 1 \pmod 4$. Factor the left-hand side as
\begin{equation}\label{eq:factor}
X^{3}+6 = (X+2)(X^{2}-2X+4).
\end{equation}
Because $X\equiv 1 \pmod 4$, both factors on the right-hand side of
\eqref{eq:factor} are odd and congruent to $3$ modulo $4$.

A short gcd computation gives
\begin{align*}
\gcd(X+2, X^{2}-2X+4)
&= \gcd(X+2, (X^{2}-2X+4) - (X-4)(X+2)) \\
&= \gcd(X+2, 12).
\end{align*}
Since $X\equiv 1 \pmod 4$, $X+2 \equiv 3 \pmod 4$. If $3 \mid X+2$, then
$X\equiv 1 \pmod 3$ and therefore $X^{3}+6\equiv 1 \pmod 3$. But with $z$ even
we have $z^{2}-1 \equiv 0$ or $2 \pmod 3$, and with $y$ odd we have
$y^{2} \equiv 1 \pmod 3$. Hence the right-hand side of \eqref{eq:Xeq} is
congruent to $0$ or $2$ modulo $3$, a contradiction. Thus $3\nmid X+2$, and
\begin{equation}\label{eq:gcd1}
\gcd(X+2, X^{2}-2X+4) = 1.
\end{equation}

Now compare \eqref{eq:factor} and \eqref{eq:Xeq}. The right-hand side of
\eqref{eq:Xeq} factors as
\[y^{2}(z-1)(z+1),\]
and because $z-1$ and $z+1$ are coprime, their prime divisors must be split
between the coprime factors $(X+2)$ and $(X^{2}-2X+4)$. Since $y^{2}$ is a
square and every odd square is congruent to $1$ modulo $4$, the two factors on
the right-hand side must be congruent to $1$ and $3$ modulo $4$, respectively.
By contrast, the two factors on the left-hand side of \eqref{eq:factor} are both
congruent to $3$ modulo $4$. This is impossible.

\paragraph{Conclusion.}
The four exhaustive cases all fail. Therefore there are no integers $x,y,z$ with
$z$ even, $y$ odd, and $x\equiv 3 \pmod 4$ satisfying the given equation.
