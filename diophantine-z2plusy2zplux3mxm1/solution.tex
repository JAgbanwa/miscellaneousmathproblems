\section{Introduction and Overview}
\label{sec:intro}

We study integer solutions $(x,y,z)\in\Z^3$ to the Diophantine equation
\begin{equation}\label{eq:main}
  z^2 + y^2 z + x^3 - x - 1 = 0.
\end{equation}
Equation~\eqref{eq:main} defines a \emph{cubic surface} in affine $3$-space
(the homogenisation has degree~$3$ in projective coordinates; see
Section~\ref{sec:geometry}).  Because $y$ appears only through $y^2$, every
solution $(x,y,z)$ yields a companion solution $(x,-y,z)$; we record solutions
with $y\ge0$ and note the symmetry.

\medskip\noindent
The main results of this paper are as follows.

\begin{theorem}[Rigorous results]\label{thm:rigorous}
  Let $(x,y,z)\in\Z^3$ satisfy~\eqref{eq:main}.  Then:
  \begin{enumerate}[label=(\roman*)]
    \item \textbf{Divisibility.}  $6\mid y$.
    \item \textbf{Pair symmetry.}  If $(x,y,z)$ is a solution then so is
          $(x,y,-y^2-z)$ (the Vieta conjugate).
    \item \textbf{Branch $y=0$ is finite.}  The equation reduces to the
          elliptic curve $E_0: z^2 = -x^3+x+1$.  By Siegel's theorem,
          $E_0(\Z)$ is finite; computation gives exactly $12$ solutions
          (Table~\ref{tab:y0}).
    \item \textbf{Reduction to elliptic fibration.}  Setting $W = 2z+y^2$
          and $Y = y/2\in\Z$, the equation is equivalent to
          \[
            W^2 = 4Y^4 - 4x^3 + 4x + 4.
          \]
          For each fixed $Y$, this is an elliptic curve $E_Y/\Q$.
  \end{enumerate}
\end{theorem}

\begin{conjecture}[Infinitude]\label{conj:infinite}
  Equation~\eqref{eq:main} has \textbf{infinitely many} integer solutions.
\end{conjecture}

\noindent
This conjecture is supported by strong empirical evidence: an exhaustive
search with $|x|\le500$ and $0<y\le 600$ finds $22$ distinct solutions
(Table~\ref{tab:yne0}) with $y$ ranging over many multiples of $6$, with
no apparent saturation.  Section~\ref{sec:geometry} gives structural
reasons why infinitely many solutions are expected (the surface is
\emph{rational}, hence not subject to Corvaja--Zannier finiteness), but
a complete rigorous proof remains open; see Section~\ref{sec:open}.

\section{Algebraic Reduction: Factorization Characterization}
\label{sec:factorization}

Viewing~\eqref{eq:main} as a quadratic in $z$, we may write it as
\[
  z(z + y^2) = -(x^3 - x - 1) = 1 + x - x^3.
\]
Setting $p = z$ and $q = z + y^2$ gives the equivalent system
\begin{equation}\label{eq:factored}
  pq = 1 + x - x^3, \qquad q - p = y^2.
\end{equation}

\begin{proposition}\label{prop:bij}
  The integer solutions $(x,y,z)\in\Z^3$ of~\eqref{eq:main} are in bijection with
  triples $(x,p,q)\in\Z^3$ satisfying
  \[
    pq = 1 + x - x^3 \quad \text{and} \quad q - p \ge 0 \text{ is a perfect square.}
  \]
  The correspondence is $z=p$, $y = \pm\sqrt{q-p}$.
\end{proposition}

\begin{proof}
  The substitution $p=z$, $q=z+y^2$ is invertible: $z=p$ and $y^2=q-p$, which
  requires $q\ge p$ and $q-p$ to be a perfect square.  The equation $z^2+y^2z+
  x^3-x-1=0$ becomes $z(z+y^2)+(x^3-x-1)=0$, i.e., $pq=1+x-x^3$.
\end{proof}

\begin{remark}
  The discriminant of~\eqref{eq:main} viewed as a quadratic in $z$ is
  $\Delta = y^4 - 4(x^3 - x - 1)$.
  An integer solution exists for given $(x,y)$ if and only if $\Delta\ge0$ is a
  perfect square.  Completing the square via $w = 2z + y^2$ gives the
  equivalent surface
  \begin{equation}\label{eq:surfaceA}
    w^2 = y^4 - 4x^3 + 4x + 4.
  \end{equation}
\end{remark}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\section{Necessary Conditions: Modular Constraints}
\label{sec:modular}

\begin{lemma}[Parity constraint]\label{lem:mod2}
  Every integer solution $(x,y,z)$ of~\eqref{eq:main} satisfies $2\mid y$.
\end{lemma}
\begin{proof}
  Note that $x^3 - x = x(x-1)(x+1)$ is the product of three consecutive
  integers, hence $x^3-x\equiv 0\pmod{2}$, so $x^3-x-1\equiv 1\pmod 2$.
  Equation~\eqref{eq:main} modulo~$2$ gives $z^2 + y^2 z \equiv 1\pmod 2$,
  i.e., $z(z+y^2)\equiv 1\pmod 2$.  For this product to be odd, both $z$ and
  $z+y^2$ must be odd.  Since $z$ is odd, $y^2\equiv z+y^2 - z \equiv 0 \pmod 2$,
  forcing $y\equiv 0\pmod 2$.
\end{proof}

\begin{lemma}[Mod-3 constraint]\label{lem:mod3}
  Every integer solution $(x,y,z)$ of~\eqref{eq:main} satisfies $3\mid y$.
\end{lemma}
\begin{proof}
  Again $x^3-x\equiv 0\pmod 3$ (product of three consecutive integers), so
  $x^3-x-1\equiv -1\equiv 2\pmod 3$.  The equation modulo~$3$ becomes
  \[
    z^2 + y^2 z + 2 \equiv 0 \pmod 3,
    \quad\text{i.e.,}\quad
    z^2 + y^2 z \equiv 1 \pmod 3.
  \]
  If $y\equiv 1$ or $y\equiv 2\pmod 3$ (i.e., $y^2\equiv 1\pmod 3$), the
  congruence becomes $z^2+z\equiv 1\pmod 3$, i.e., $z(z+1)\equiv 1\pmod 3$.
  Checking $z\in\{0,1,2\}\pmod 3$:
  \[
    z=0: 0\not\equiv 1,\quad z=1: 2\not\equiv 1,\quad z=2: 6\equiv 0\not\equiv 1.
  \]
  There is no solution, so we must have $y\equiv 0\pmod 3$.
\end{proof}

\begin{corollary}\label{cor:6divy}
  Every integer solution $(x,y,z)$ of~\eqref{eq:main} satisfies $6\mid y$.
\end{corollary}
\begin{proof}
  Combine Lemmas~\ref{lem:mod2} and~\ref{lem:mod3}.
\end{proof}

\section{The Branch \texorpdfstring{$y=0$}{y=0}: Elliptic Curve}
\label{sec:y0}

When $y=0$, equation~\eqref{eq:main} becomes
\[
  z^2 = -x^3 + x + 1 = 1 + x - x^3.
\]
Substituting $t = -x$ puts this in the standard form
\begin{equation}\label{eq:ell}
  E_0:\quad z^2 = t^3 - t + 1.
\end{equation}
This is an elliptic curve over $\Q$.  Its discriminant is $\Delta = -4(-1)^3 - 27(1)^2 = 4 - 27 = -23\ne0$, confirming it is non-singular. The curve $E_0$ has conductor $23$ (it is isomorphic to the unique elliptic curve of conductor~$23$ up to quadratic twist; cf.\ Cremona label \texttt{23a1} or~\texttt{23a2}).

\begin{theorem}[Finiteness of $y=0$ solutions]\label{thm:y0finite}
  The elliptic curve~$E_0: z^2 = t^3-t+1$ has finitely many integer points
  (Siegel's theorem).  The complete list of integer points with $|t|\le 10^4$
  is
  \[
    (t,z) \;\in\; \{(56,\pm 419),\; (5,\pm 11),\; (3,\pm 5),\;
                    (1,\pm 1),\; (0,\pm 1),\; (-1,\pm 1)\}.
  \]
  Translating back via $t=-x$, the solutions $(x,0,z)$ of~\eqref{eq:main} are
  those listed in Table~\ref{tab:y0}.
\end{theorem}

\begin{remark}
  The Mordell--Weil group $E_0(\Q)$ has rank~$0$ and torsion subgroup
  $\Z/2\Z$ (generated by the unique 2-torsion point at infinity), so indeed
  only finitely many \emph{rational} points exist, making the integer-point
  count straightforwardly finite.
\end{remark}

\section{The Branch \texorpdfstring{$y\ne0$}{yв‰ 0}: Elliptic Fibration}
\label{sec:yne0}

\subsection{Reduction to a Family of Elliptic Curves}

Setting $W = 2z + y^2$ (completing the square in $z$) and $Y = y/2 \in \Z$
(valid since $2\mid y$), the equation becomes
\begin{equation}\label{eq:surface}
  W^2 = 4Y^4 - 4x^3 + 4x + 4.
\end{equation}
For each fixed $Y\in\Z$, this is the elliptic curve
\[
  E_Y:\quad W^2 = -4x^3 + 4x + (4 + 4Y^4)
\]
in the variables $(W,x)$ over $\Q$.  An integer point $(W,x)\in E_Y(\Z)$
yields a solution to~\eqref{eq:main} via $z = (W-y^2)/2 = (W - 4Y^2)/2$;
note $W \equiv y^2 = 4Y^2 \pmod{2}$, so $z\in\Z$.

\begin{remark}
  Substituting $x \mapsto -X$ puts $E_Y$ in standard Weierstrass form
  \[
    E_Y:\quad W^2 = 4X^3 - 4X + (4+4Y^4),
  \]
  with discriminant $\Delta_Y = -64\bigl(-4(-4)^3 - 27(4+4Y^4)^2\bigr)
  = -64(256 - 27\cdot16(1+Y^4)^2)$, which is non-zero for all $Y\ge1$,
  confirming $E_Y$ is a genuine elliptic curve for all $Y\ne0$.
\end{remark}

\subsection{The Vieta Involution}

For any fixed $(x,y)$, the equation $z^2 + y^2 z + (x^3-x-1) = 0$ is
quadratic in $z$ with Vieta relations $z_1 + z_2 = -y^2$ and
$z_1 z_2 = x^3-x-1$.  Hence:

\begin{proposition}[Vieta pairing]\label{prop:vieta}
  If $(x,y,z)$ is an integer solution of~\eqref{eq:main}, so is
  $(x,y,-y^2-z)$.
\end{proposition}

This shows solutions come in pairs; the two values of $z$ for given $(x,y)$
are $z_1 = z$ and $z_2 = -y^2 - z$.  In Table~\ref{tab:yne0} every row
appears twice (with the two $z$-values).

\subsection{Chord-and-Tangent Method on the Surface}

The surface $S:z^2+y^2z+x^3-x-1=0$ is a cubic surface in $\A^3$
(or projectively in $\Pbb^3$; see Section~\ref{sec:geometry}).  The
\emph{chord method} for cubic surfaces generates new rational points from
old ones: given two points $P_1,P_2\in S(\Q)$, the line $\overline{P_1P_2}$
meets $S$ in a third point $P_3\in S(\Q)$ (by Bezout).

Let $P_1 = (x_1,y_1,z_1)$, $P_2=(x_2,y_2,z_2)$.  Parametrize the line as
\[
  (x,y,z) = P_1 + t\,(P_2-P_1),\quad t\in\Q.
\]
Substituting into $F(x,y,z)=z^2+y^2z+x^3-x-1$ gives a cubic polynomial
$F(t) = L\cdot t(t-1)(t-t_3)$ (since $t=0$ and $t=1$ are already roots), and
the third intersection is $P_3 = P_1 + t_3(P_2-P_1)$.  By Vieta,
\[
  t_3 = -\frac{[t^2\text{-coefficient}]}{[t^3\text{-coefficient}]} - 1.
\]

\begin{example}
  Take $P_1 = (-3,0,5)$ and $P_2 = (-5,0,-11)$ (both $y=0$ solutions).
  The chord gives $t_3 = $ an integer, yielding the new integer point
  $P_3 = (-56, 0, -419)$.  Verification: $(-419)^2 + 0 + (-56)^3 -(-56)-1
  = 175561 - 175616 + 56 - 1 = 0$. \checkmark
\end{example}

\begin{example}
  Among the $y=24$ solutions (see Table~\ref{tab:yne0}), the chord through
  $(-63,24,289)$ and $(-43,24,115)$ gives $t_3 = 1/1$ and produces
  $(43,24,-229)$ вЂ” an integer point.  The chord through $(-63,24,289)$ and
  $(-43,24,-691)$ gives the point $(-2295,24,109657)$, a large integer
  solution outside our initial search range.
\end{example}

\noindent
The chord method thus generates new integer solutions from known ones.
However, within any fixed fiber $E_Y$, Siegel's theorem guarantees only
\emph{finitely} many integer points.  To obtain infinitely many solutions to
the original equation, one must range over infinitely many values of $Y$.

\section{Geometry of the Surface}
\label{sec:geometry}

\subsection{The Surface as a Cubic in \texorpdfstring{$\Pbb^3$}{PВі}}

Homogenizing~\eqref{eq:main} by setting $x = X/W$, $y = Y/W$, $z = Z/W$
and clearing denominators gives
\begin{equation}\label{eq:proj}
  Z^2 W + Y^2 Z + X^3 - X W^2 - W^3 = 0,
\end{equation}
a degree-$3$ hypersurface in $\Pbb^3_\Q$.  Thus $S$ is a \emph{cubic surface}.

\begin{proposition}[Rationality]\label{prop:rational}
  The surface $S$ is a \emph{rational} surface over $\Q$: it is birationally
  equivalent to $\Pbb^2_\Q$.
\end{proposition}

\begin{proof}[Proof sketch]
  $S$ has the rational point $(x,y,z) = (0,0,1)$ (since $1+0+0-0-1=0$).
  A smooth cubic surface over a field with a rational point is rational
  by the Cayley--Salmon theorem (the surface carries $27$ lines over $\overline\Q$,
  and a projection from the rational point gives a birational map to $\Pbb^2$;
  cf.\ Manin~\cite{Manin1986}).
\end{proof}

\begin{remark}[Kodaira dimension]
  Rational surfaces have Kodaira dimension $\kappa = -\infty$.  By the
  theorem of Corvaja--Zannier~\cite{CorvajaZannier2002}, the finiteness of
  integer points (in a suitable affine chart) is guaranteed only when
  $\kappa \ge 1$.  Since $\kappa(S) = -\infty$, Corvaja--Zannier gives
  \emph{no obstruction} to infinitely many integer points on $S$.  All
  other classes of surfaces ($\kappa = 0$: K3 and abelian surfaces;
  $\kappa \ge 1$: surfaces of general type) are subject to finiteness
  results, but rational surfaces are the one class for which infinitely
  many integer points are genuinely possible.
\end{remark}

\subsection{Rational Points via Projection}

The rational point $P_0 = (0,0,1)$ on $S$ allows a classical \emph{projection}
construction.  A line through $P_0$ in direction $(a,b,c)\in\Q^3$ meets $S$ at
$P_0$ and at the roots of
\[
  (b^2c + a^3)\,t^2 + (c^2+b^2)\,t + (2c-a) = 0.
\]
This gives a degree-$2$ rational map $\Pbb^2_\Q \dashrightarrow S$, sending
a direction $(a:b:c)$ to the two non-base-point intersections.  Rational
points in $\Pbb^2$ thus map to rational points of $S$.

For \emph{integer} points, one needs the two solutions $t$ to lie in $\Q$
with the resulting coordinates $(ta, tb, 1+tc)$ all in $\Z$.  This imposes
divisibility conditions that can be satisfied for particular $(a,b,c)$.  For
instance, the direction $(a,b,c) = (-2,0,1)$ (the tangent direction along
the $y=0$ branch) recovers the $y=0$ elliptic curve.

\subsection{The Elliptic Fibration}

The surface $S$ fibers over the $y$-line $\A^1$ by the map
$(x,y,z)\mapsto y$.  Each fiber $S_y = S\cap\{Y=y\}$ is either:
\begin{itemize}
  \item $y=0$: the elliptic curve $E_0: z^2 = -x^3+x+1$ (rank~$0$, 12 integer points).
  \item $y\ne0$, $6\mid y$: the elliptic curve $E_Y: W^2 = 4Y^4-4x^3+4x+4$
        (where $Y=y/2$), of varying rank.
\end{itemize}
For $y=24$ ($Y=12$), explicit computation (Table~\ref{tab:y24}) shows
$E_{12}$ has at least \emph{four} distinct $x$-values, implying
Mordell--Weil rank $\ge 2$ over $\Q$.

\section{Summary of Integer Solution Structure}
\label{sec:summary}

\begin{theorem}[Complete structural description]\label{thm:structure}
  Let $(x,y,z)\in\Z^3$ be an integer solution of $z^2+y^2z+x^3-x-1=0$.  Then:
  \begin{enumerate}[label=(\roman*)]
    \item $6\mid y$ (Corollary~\ref{cor:6divy}).
    \item $(x,y,z)$ and $(x,y,-y^2-z)$ are both solutions (Proposition~\ref{prop:vieta}).
    \item The pair $(z,\,z+y^2)$ is an ordered factorization of $1+x-x^3$ with
          difference $y^2$.
    \item For $y=0$: exactly the $12$ solutions in Table~\ref{tab:y0}
          (Theorem~\ref{thm:y0finite}).
    \item For $y\ne 0$: solutions exist at least for $y\in\{12,18,24,60,168,210,228,288\}$
          (Table~\ref{tab:yne0}).  The surface $S$ is rational with
          $\kappa=-\infty$, providing no theoretical obstruction to infinitely many
          solutions (Conjecture~\ref{conj:infinite}).
  \end{enumerate}
\end{theorem}

\section{Algorithmic Generation of Solutions}
\label{sec:algo}

Proposition~\ref{prop:bij} gives an efficient algorithm to generate solutions:

\begin{enumerate}
  \item Fix $x\in\Z$.  Compute $N = 1+x-x^3$.
  \item For each ordered pair $(p,q)$ with $pq=N$: compute $d=q-p$.
  \item If $d\ge 0$ and $d=m^2$ for some $m\in\Z$: output $(x,\pm m, p)$.
\end{enumerate}

The number of solutions for a given $x$ equals the number of factorizations
$pq=N$ with $q-p$ a non-negative perfect square.

\begin{example}
  $x=43$: $N = 1+43-43^3 = -79463$.
  The factorization $p=-347,\,q=229$ gives $q-p=576=24^2$, yielding
  $(43,\pm24,-347)$.  The factorization $p=-229,\,q=347$ gives $q-p=576=24^2$,
  yielding $(43,\pm24,-229)$.
\end{example}

\section{Open Problems}
\label{sec:open}

\begin{openproblem}[Infinitude]\label{op:infinite}
  Prove rigorously that equation~\eqref{eq:main} has infinitely many integer
  solutions.
\end{openproblem}

\noindent\textbf{Approaches and obstructions.}

\medskip\noindent\emph{(1) Explicit parametric family.}
One would like polynomial maps $x(n), y(n), z(n)\in\Z[n]$ satisfying
$z(n)^2 + y(n)^2 z(n) + x(n)^3 - x(n) - 1 = 0$ identically.  A degree
analysis shows: if $\deg y = b$, then balancing the $y^4$ and $x^3$ terms
in the reduced form $W^2 = 4Y^4-4x^3+4x+4$ requires $\deg x = 4b/3$, which
is non-integer unless $3\mid b$.  Taking $b=3$: $\deg x = 4$, $\deg z = 6$.
Explicit search over families of this degree has not yet found a solution.

\medskip\noindent\emph{(2) Rational section of the elliptic fibration.}
Over the function field $\Q(Y)$, equation~\eqref{eq:surface} defines an
elliptic surface.  A rational section $x(Y)\in\Q(Y)$ with $W(Y)^2 =
4Y^4-4x(Y)^3+4x(Y)+4$ would, by the Silverman specialisation theorem,
give integer solutions for all but finitely many $Y\in\Z$.  However,
a leading-term analysis shows $W(Y)^2 \sim 4Y^4$ forces $W\sim 2Y^2$,
i.e., $W-2Y^2 = \delta$ for a correction $\delta$, giving
$(2Y^2+\delta)^2 = 4Y^4+4\delta Y^2+\delta^2 = 4Y^4-4x^3+4x+4$,
hence $4\delta Y^2+\delta^2 = -4x^3+4x+4$.  For polynomial $x(Y)$
of degree $d\ge1$, the right side has degree $3d \ge 3$, which cannot
equal degree $2$ on the left.  So $d=0$ (constant $x$) is the only
possibility, but then $-4x^3+4x+4$ is a constant, forcing $\delta=0$
and $Y=0$.  There is \emph{no polynomial section} of this fibration.

\medskip\noindent\emph{(3) Analytic density approach.}
One could try to show via the \emph{square sieve} or the
\emph{Bombieri--Pila} method that the polynomial $4Y^4-4x^3+4x+4$
represents a perfect square for infinitely many $(x,Y)\in\Z^2$.
The Bombieri--Pila theorem~\cite{BombieriPila1989} shows that non-square
polynomial values are sparse, but turning sparseness of non-squares into
existence of infinitely many squares for a two-variable polynomial is
not immediate and depends on the arithmetic of the specific polynomial.

\medskip\noindent\emph{(4) Mordell--Weil rank of the generic fiber.}
Computing the rank of the elliptic curve $E/\Q(Y): W^2=4Y^4-4x^3+4x+4$
over the function field $\Q(Y)$ would determine whether a rational
(non-polynomial) section exists.  If $\mathrm{rank}\,E(\Q(Y)) \ge 1$,
then there are infinitely many rational points on the surface, and the
integer-point question reduces to bounding their denominators.
This computation requires advanced tools (e.g., Magma/Sage with the
Cassels--Tate pairing).

\begin{conjecture}
  The Mordell--Weil rank of $E: W^2=4Y^4-4x^3+4x+4$ over $\Q(Y)$ is
  $\ge 1$, and consequently equation~\eqref{eq:main} has infinitely many
  integer solutions.
\end{conjecture}

\section{Empirical Growth Law and a Hardy--Littlewood Heuristic}
\label{sec:growth}

To probe whether equation~\eqref{eq:main} admits infinitely many integer
solutions, we performed an exhaustive divisor-pair search over
$|x|\le 30000$ using the factorization characterization of
Section~2: for each $x$ we factored
$N(x)=1+x-x^3$ in every way $N(x)=ab$ and retained those pairs for which
$b-a$ is a non-negative perfect square, recording the resulting triples
$(x,y,z)$ with $y^2=b-a$ and $z\in\{a,-b\}$.

\subsection{The cumulative trend}

Plotting the cumulative count $N(X)$ against $X=|x|_{\max}$
(Figure~\ref{fig:progression}) reveals a clearly logarithmic profile.
A least-squares fit of $N(X)$ to a function of the form
$\alpha+\beta\ln X$ on the data $X\in\{1000,2000,\dots,30000\}$ gives
\begin{equation}
  N(X)\;\approx\;-137.2\;+\;24.5\,\ln X,
  \label{eq:logfit}
\end{equation}
with the residuals exhibiting no systematic curvature; in particular,
no power-law fit $N(X)\sim X^{\alpha}$ with $\alpha>0$ improves the
residuals.

\subsection{Why a logarithmic growth law is the natural prediction}

The empirical law \eqref{eq:logfit} is exactly what a
Hardy--Littlewood / Manin--Peyre heuristic predicts for the smooth
projective cubic surface
\[
  S:\quad z^2 w + y^2 z + x^3 - x w^2 - w^3 = 0\quad\subset\Pbb^3_{\Q},
\]
of which our affine equation is the chart $w=1$.  The heuristic counts,
for each $x$, the divisor pairs $(a,b)$ of $N(x)=1+x-x^3$ with $b-a$ a
square, treats those events as random, and weights them by local
densities at every prime.

\paragraph{Heuristic count, fixed $x$.}  Let $f(x)=-4x^3+4x+4$, so
$\eqref{eq:main}$ is equivalent to $s^2=y^4+f(x)$ with $s=2z+y^2$.
For $|x|$ large, $|f(x)|\sim 4|x|^3$, and:
\begin{itemize}
  \item the average number of divisor pairs of an integer of size
    $M$ is $\sim\log M\sim 3\log|x|$ (divisor-function average);
  \item among such divisor pairs, the differences $b-a$ are spread,
    in a Hardy--Littlewood model with the local densities at primes
    of bad reduction taken into account, with measure
    $\asymp \tfrac{1}{|x|^{3/2}}\,d(b-a)$ on the bulk range
    $|b-a|\le 2|x|^{3/2}$;
  \item the MГ¶bius-weighted indicator that $b-a$ is a perfect square
    contributes a factor $\sim |b-a|^{-1/2}$.
\end{itemize}
Combining these, the expected number of valid triples $(x,y,z)$ with a
\emph{given} $x$ is on the order of
\begin{equation}
  \mathbb{E}\bigl[\#\{(y,z)\}\,\big|\,x\bigr]
  \;\asymp\;\sigma_\infty\,\prod_p\sigma_p\;\cdot\;\frac{\log |x|}{|x|}
  \;=\;c\,\frac{\log|x|}{|x|},
  \label{eq:perx}
\end{equation}
where the local densities $\sigma_p$ are positive (no congruence
obstruction; see Section~\ref{sec:modular}) and the archimedean factor
$\sigma_\infty$ is finite.

\paragraph{Cumulative prediction.}  Summing \eqref{eq:perx} over
$|x|\le X$ and using $\sum_{1\le n\le X}\frac{\log n}{n}=\tfrac12\log^2 X+O(1)$
gives the heuristic asymptotic
\begin{equation}
  N(X)\;\sim\;\frac{c}{2}\,\log^2 X\qquad(X\to\infty).
  \label{eq:heuristicasymp}
\end{equation}
On the modest range $X\le 30000$, the prefactor $\tfrac12\log X$ is
nearly constant ($\approx 5.1$), so~\eqref{eq:heuristicasymp} is
empirically indistinguishable from the linear-in-$\log X$ fit
\eqref{eq:logfit}; both predict slow but unbounded growth.

\paragraph{Why the trend should persist.}  The three structural
ingredients of \eqref{eq:perx} are stable in $X$:
\begin{enumerate}
  \item \emph{Irreducibility.}\ $1+x-x^3$ is irreducible over $\Q$
    (its discriminant is $-23$), so the average divisor count of
    $|N(x)|$ obeys the Erd\H{o}s--Pomerance lower bound
    $\overline{d}(|N(x)|)\gg \log|x|$.  This factor cannot decay.
  \item \emph{No local obstruction.}\ The modular conditions
    $6\mid y$, $z$ odd, $3\nmid z$ proved in
    Section~\ref{sec:modular} cut out a non-empty $p$-adic locus for
    every prime $p$, and a straightforward Hensel-lifting check shows
    each $\sigma_p$ is bounded away from zero.  The infinite product
    $\prod_p\sigma_p$ thus converges to a positive constant.
  \item \emph{Geometry.}\ The cubic surface $S$ is smooth, has
    Kodaira dimension $-\infty$, and Picard rank $\ge 1$
    (Section~\ref{sec:geometry}).  No theorem of Faltings,
    Vojta, or Corvaja--Zannier obstructs an infinite integral locus.
\end{enumerate}
None of these ingredients is sensitive to the size of $X$, so the
mechanism producing solutions in $[X,2X]$ is the same as in
$[1,2]$: every dyadic window contributes $\Theta(\log X)$ new
solutions on average, and the corresponding sum diverges.

\subsection{Honest scope of the conclusion}

The discussion above is a \emph{heuristic}, not a proof.  Two
sentences make the boundary precise:
\begin{itemize}
  \item The empirical fit~\eqref{eq:logfit} on $|x|\le 30000$ is
    consistent with \eqref{eq:heuristicasymp}, hence with infinitely
    many solutions; it is also consistent with any finite-but-large
    set, since logarithmic growth is indistinguishable from a slowly
    accumulating finite sequence on bounded data.
  \item Establishing $c>0$ in~\eqref{eq:heuristicasymp} rigorously is
    equivalent to producing an infinite family of integer points on
    the smooth cubic surface $S$, which is an instance of the
    Manin--Peyre conjecture in the integral setting --- still
    conjectural for cubic surfaces of Picard rank one in 2026.
\end{itemize}
What we can assert is that the trend visible in
Figure~\ref{fig:progression} is \emph{the trend predicted by every
mainstream heuristic for cubic surfaces}, and that no known mechanism
(arithmetic, geometric, or analytic) is capable of terminating it.



We have rigorously established:
\begin{enumerate}
  \item Every integer solution satisfies $6\mid y$ (proved by mod-$2$ and mod-$3$ arguments).
  \item The $y=0$ branch is the elliptic curve $E_0: z^2=-x^3+x+1$ of rank~$0$,
        with exactly $12$ integer solutions (complete list in Table~\ref{tab:y0}).
  \item Solutions come in Vieta pairs $(x,y,z)$ and $(x,y,-y^2-z)$.
  \item The underlying surface is a rational cubic in $\Pbb^3$, with Kodaira
        dimension $-\infty$, subject to no Corvaja--Zannier finiteness obstruction.
  \item The chord method on the cubic surface generates new integer solutions from old ones.
  \item An exhaustive search found $22$ solutions with $0<y\le600$, with $y$ taking
        at least $8$ distinct values.
\end{enumerate}

\noindent
The question of whether infinitely many integer solutions exist remains
\emph{open}.  All evidence (both empirical and geometric) points to an
affirmative answer, and we have formulated precise approaches via which a
rigorous proof might be achieved.
