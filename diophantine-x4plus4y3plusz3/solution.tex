\section{Statement and Main Result}\label{sec:intro}

\subsection{The equation}

We study the Diophantine equation
\begin{equation}\label{eq:main}
  x^4 + 4y^3 + z^3 = 0, \qquad (x,y,z)\in\Z^3. \tag{$\star$}
\end{equation}

\subsection{Main theorem}

\begin{theorem}[Main result]\label{thm:main}
  The equation $x^4 + 4y^3 + z^3 = 0$ has \textbf{infinitely many} integer solutions.
  Four primary parametric families (derived by elementary substitutions) are
  \begin{equation}\label{eq:family1}
    (x, y, z) = \bigl(t^3,\; 0,\; -t^4\bigr),
    \qquad t \in \Z,
  \end{equation}
  \begin{equation}\label{eq:family2}
    (x, y, z) = \bigl(3t^3,\; -3t^4,\; 3t^4\bigr),
    \qquad t \in \Z,
  \end{equation}
  \begin{equation}\label{eq:family3}
    (x, y, z) = \bigl(-4t^3,\; -4t^4,\; 0\bigr),
    \qquad t \in \Z,
  \end{equation}
  \begin{equation}\label{eq:family4}
    (x, y, z) = \bigl(-5t^3,\; -5t^4,\; -5t^4\bigr),
    \qquad t \in \Z.
  \end{equation}
  An additional three families $(4t^3, 4t^4, -8t^4)$,
  $(12t^3, -12t^4, -24t^4)$, and $(-5t^3, -10t^4, 15t^4)$
  are discovered computationally (Section~\ref{sec:extra}).
\end{theorem}

The proofs are elementary: each family is verified by a single ring identity,
and injectivity is established by strict monotonicity of $t \mapsto t^3$ on $\N$.
See Section~\ref{sec:structure} for the weighted-homogeneity structure,
Section~\ref{sec:derivation} for derivations of each family,
Section~\ref{sec:proof} for the complete proof,
Section~\ref{sec:explicit} for a table of small solutions,
and Section~\ref{sec:lean} for the Lean~4 formalisation.

\section{Weighted-Homogeneity Structure}\label{sec:structure}

\begin{definition}[Weighted degree]
  Assign weights $\mathrm{wt}(x) = 3$, $\mathrm{wt}(y) = 4$, $\mathrm{wt}(z) = 4$.
  Then every term of~\eqref{eq:main} has \emph{weighted degree} $12$:
  \[
    \mathrm{wt}(x^4) = 4 \cdot 3 = 12, \qquad
    \mathrm{wt}(4y^3) = 3 \cdot 4 = 12, \qquad
    \mathrm{wt}(z^3) = 3 \cdot 4 = 12.
  \]
\end{definition}

\begin{lemma}[Weighted scaling invariance]\label{lem:scale}
  If $(a, b, c) \in \Z^3$ satisfies~\eqref{eq:main}, then so does
  $(t^3 a,\; t^4 b,\; t^4 c)$ for every $t \in \Z$.
\end{lemma}

\begin{proof}
  Substitute directly:
  \[
    (t^3 a)^4 + 4(t^4 b)^3 + (t^4 c)^3
    = t^{12}(a^4 + 4b^3 + c^3) = t^{12} \cdot 0 = 0. \qquad\qed
  \]
\end{proof}

\begin{corollary}
  To produce an infinite parametric family of integer solutions it suffices to
  exhibit a single non-trivial \emph{seed} solution $(a, b, c)$.
\end{corollary}

\begin{proof}
  Apply Lemma~\ref{lem:scale}.  The map $t \mapsto (t^3 a, t^4 b, t^4 c)$ is
  injective on $\N$ because $t \mapsto t^3$ is strictly monotone.
\end{proof}

\begin{remark}
  The weight assignment $(3, 4, 4)$ with degree $12$ contrasts with the weights
  $(3, 5, 5)$ and degree $15$ of $x^5 + 2y^3 + z^3 = 0$, and the uniform weights
  $(1,1,1)$ and degree $5$ of $4x^5 + 4y^5 + 11z^5 = 0$.  The even exponent of $x$
  means that $x$ and $-x$ make the same contribution $x^4$, so sign changes in $x$
  alone do not produce new solutions.
\end{remark}

\section{Derivation of the Families}\label{sec:derivation}

All four families exploit the same core algebraic observation:
\begin{equation}\label{eq:core}
  s^4 = r^3 \quad \text{is solved by} \quad s = t^3,\; r = t^4 \quad \text{for any } t \in \Z.
\end{equation}

\subsection{Family~1: setting $y = 0$}\label{sec:fam1}

Substituting $y = 0$ into~\eqref{eq:main} gives
\[
  x^4 + z^3 = 0, \qquad \text{i.e.} \qquad z^3 = -x^4.
\]
Integer solutions require $x^4$ to be a perfect cube.  Setting $x = t^3$
gives $x^4 = t^{12}$, so $z^3 = -t^{12}$ and $z = -t^4$.

\textbf{Family~1:} $(x, y, z) = (t^3,\; 0,\; -t^4)$, $t \in \Z$.

\textbf{Direct verification:}
\[
  (t^3)^4 + 4 \cdot 0^3 + (-t^4)^3 = t^{12} + 0 - t^{12} = 0.\quad\checkmark
\]

\textbf{Seed:} $(1, 0, -1)$: $1 + 0 - 1 = 0$. $\checkmark$

\subsection{Family~2: setting $z = -y$}\label{sec:fam2}

Substituting $z = -y$ into~\eqref{eq:main} gives
\[
  x^4 + 4y^3 + (-y)^3 = x^4 + 3y^3 = 0.
\]
Setting $x = 3s$ and $y = -3r$:
\[
  81s^4 + 3(-3r)^3... \quad\text{better:}\quad
  (3s)^4 + 3(-3r)^3 = 81s^4 - 81r^3 = 81(s^4 - r^3) = 0.
\]
So $s^4 = r^3$; by~\eqref{eq:core}, set $s = t^3$ and $r = t^4$, giving
$x = 3t^3$ and $y = -3t^4$, $z = 3t^4$.

\textbf{Family~2:} $(x, y, z) = (3t^3,\; -3t^4,\; 3t^4)$, $t \in \Z$.

\textbf{Direct verification:}
\[
  (3t^3)^4 + 4(-3t^4)^3 + (3t^4)^3
  = 81t^{12} - 108t^{12} + 27t^{12} = 0.\quad\checkmark
\]

\textbf{Seed:} $(3, -3, 3)$: $81 - 108 + 27 = 0$. $\checkmark$

\subsection{Family~3: setting $z = 0$}\label{sec:fam3}

Substituting $z = 0$ into~\eqref{eq:main} gives
\[
  x^4 + 4y^3 = 0, \qquad \text{i.e.} \qquad x^4 = -4y^3.
\]
Setting $x = -4s$ and $y = -4r$:
\[
  256s^4 = 256r^3, \qquad \text{so} \qquad s^4 = r^3.
\]
By~\eqref{eq:core}, $s = t^3$ and $r = t^4$, giving $x = -4t^3$ and $y = -4t^4$.

\textbf{Family~3:} $(x, y, z) = (-4t^3,\; -4t^4,\; 0)$, $t \in \Z$.

\textbf{Direct verification:}
\[
  (-4t^3)^4 + 4(-4t^4)^3 + 0^3
  = 256t^{12} - 256t^{12} = 0.\quad\checkmark
\]

\textbf{Seed:} $(-4, -4, 0)$: $256 - 256 + 0 = 0$. $\checkmark$

\subsection{Family~4: setting $y = z$}\label{sec:fam4}

Substituting $y = z$ into~\eqref{eq:main} gives
\[
  x^4 + 5y^3 = 0, \qquad \text{i.e.} \qquad x^4 = -5y^3.
\]
Setting $x = -5t^3$ and $y = -5t^4$:
\[
  (-5t^3)^4 = 625t^{12} \qquad \text{and} \qquad -5(-5t^4)^3 = -5 \cdot (-125t^{12}) = 625t^{12}.
\]
Hence $x^4 = -5y^3$. $\checkmark$

\textbf{Family~4:} $(x, y, z) = (-5t^3,\; -5t^4,\; -5t^4)$, $t \in \Z$.

\textbf{Direct verification:}
\[
  (-5t^3)^4 + 4(-5t^4)^3 + (-5t^4)^3
  = 625t^{12} - 500t^{12} - 125t^{12} = 0.\quad\checkmark
\]

\textbf{Seed:} $(-5, -5, -5)$: $625 - 500 - 125 = 0$. $\checkmark$

\subsection{Additional seeds found computationally}\label{sec:extra}

A brute-force search over $|x|,|y|,|z|\leq 50$ (see \texttt{brute\_force\_search.py})
reveals three further infinite families:

\begin{itemize}
  \item \textbf{Family~5: $(4t^3, 4t^4, -8t^4)$.}
    Setting $y = -x$, $z = 2x$ reduces the equation to $x^3(x+4) = 0$,
    giving seed $(-4, 4, -8)$.
    \[
      (4t^3)^4 + 4(4t^4)^3 + (-8t^4)^3
      = 256t^{12} + 256t^{12} - 512t^{12} = 0.\quad\checkmark
    \]

  \item \textbf{Family~6: $(12t^3, -12t^4, -24t^4)$.}
    Setting $z = 2y$ reduces the equation to $x^4 + 12y^3 = 0$,
    solved by $x = 12t^3$, $y = -12t^4$.
    \[
      (12t^3)^4 + 4(-12t^4)^3 + (-24t^4)^3
      = 20736t^{12} - 6912t^{12} - 13824t^{12} = 0.\quad\checkmark
    \]

  \item \textbf{Family~7: $(-5t^3, -10t^4, 15t^4)$.}
    Setting $y = 2x$, $z = -3x$ reduces the equation to $x^3(x+5) = 0$,
    giving seed $(-5, -10, 15)$.
    \[
      (-5t^3)^4 + 4(-10t^4)^3 + (15t^4)^3
      = 625t^{12} - 4000t^{12} + 3375t^{12} = 0.\quad\checkmark
    \]
\end{itemize}

All 19 solutions found in the search range $|x|,|y|,|z|\leq 50$ are accounted for
by Families~1--7 (see the tables in Sections~\ref{sec:fam1}--\ref{sec:fam4} and above).

\section{Proof of the Main Theorem}\label{sec:proof}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
  All four ring identities are verified in Sections~\ref{sec:fam1}--\ref{sec:fam4}.

  \textbf{(Infinitude of solutions via Family~1.)}
  Consider the map
  \[
    \varphi : \N \to \Z^3, \qquad \varphi(n) = (n^3,\; 0,\; -n^4).
  \]
  Each $\varphi(n)$ is in the solution set by the verification in Section~\ref{sec:fam1}.
  For $n_1 < n_2$ (both non-negative integers), we have $n_1^3 < n_2^3$
  (strict monotonicity of cubing on $\N$), so $n_1^3 \neq n_2^3$, hence
  $\varphi(n_1) \neq \varphi(n_2)$.  Therefore $\varphi$ is injective and the
  image is an infinite subset of the solution set.
  \qed
\end{proof}

\begin{lstlisting}[language=Lean4,caption={Injectivity and infinitude.}]
def natMap (n : N) : Z x Z x Z := ((n : Z) ^ 3, 0, -((n : Z) ^ 4))

theorem natMap_mem (n : N) : natMap n in solutions := by
  simp only [solutions, Set.mem_setOf_eq, isolution, natMap]; ring

private theorem pow3_strictMono : StrictMono (fun n : N => n ^ 3) := by
  intro a b hab; exact Nat.pow_lt_pow_left hab (by norm_num)

theorem natMap_injective : Function.Injective natMap := by
  intro n1 n2 h
  simp only [natMap, Prod.mk.injEq] at h
  have h3 : (n1 : Z) ^ 3 = (n2 : Z) ^ 3 := h.1
  have h3n : n1 ^ 3 = n2 ^ 3 := by exact_mod_cast h3
  exact pow3_strictMono.injective h3n

theorem solutions_infinite : solutions.Infinite := by
  apply (Set.infinite_range_of_injective natMap_injective).mono
  rintro _ <n, rfl>
  exact natMap_mem n
\end{lstlisting}

\section{Summary}\label{sec:summary}

\begin{theorem}[Complete statement]
  The Diophantine equation $x^4 + 4y^3 + z^3 = 0$ has \textbf{infinitely many}
  integer solutions.  Seven explicit infinite parametric families are:
  \begin{align*}
    \text{Family 1:} &\quad (t^3,\; 0,\; -t^4), & t \in \Z, \\
    \text{Family 2:} &\quad (3t^3,\; -3t^4,\; 3t^4), & t \in \Z, \\
    \text{Family 3:} &\quad (-4t^3,\; -4t^4,\; 0), & t \in \Z, \\
    \text{Family 4:} &\quad (-5t^3,\; -5t^4,\; -5t^4), & t \in \Z, \\
    \text{Family 5:} &\quad (4t^3,\; 4t^4,\; -8t^4), & t \in \Z, \\
    \text{Family 6:} &\quad (12t^3,\; -12t^4,\; -24t^4), & t \in \Z, \\
    \text{Family 7:} &\quad (-5t^3,\; -10t^4,\; 15t^4), & t \in \Z. \\
  \end{align*}
  Each family is verified by a single ring identity.  Infinitude follows from
  the injectivity of $n \mapsto (n^3, 0, -n^4)$ on $\N$.
  A complete, unconditional Lean~4 proof with zero sorry and zero additional axioms
  is given in \texttt{InfinitelyManySolutions.lean}.
\end{theorem}
